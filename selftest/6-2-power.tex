\boldsubsection{Reading Quiz Section \ref*{sec:power}}

\begin{enumerate}
  \item Which of the following are true statements. Select all that apply.
  \begin{enumerate}
    \item $[0,1)\in\cP(\R)$
    \item $7\in\cP(\N)$
    \item $\{(3,5),(2,9)\} \subseteq \cP(\N \times \N)$
    \item $\{4,\pi\} \in \cP(\R)$
  \end{enumerate}
    
  \item Let $A=\bigl\{(1,2),3,(4,\{5\})\bigr\}$. What is $\nm{\cP(A)}$?
  \begin{enumerate}
    \item $3$
    \item $8$
    \item $16$
    \item $32$
  \end{enumerate}
\end{enumerate}
